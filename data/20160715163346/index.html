# by Florian Beer
# http://blog.no-panic.at/2015/04/21/set-tmux-pane-title-on-ssh-connections/
#
# I am frequently connecting to several servers over SSH at once 
# and like to use tmux to organize those different sessions. 
# To be able to keep track which pane is which server I am using 
# the following function in my .bashrc file to display the machine
# name in tmux’s pane title.

# https://gist.github.com/florianbeer/ee02c149a7e25f643491
ssh() {
    if [ "$(ps -p $(ps -p $$ -o ppid=) -o comm=)" = "tmux" ]; then
        tmux rename-window "$(echo $* | cut -d . -f 1)"
        command ssh "$@"
        tmux set-window-option automatic-rename "on" 1>/dev/null
    else
        command ssh "$@"
    fi
}

    Actually, what changed by the above script changed is just the window title. If you have two panes in the window, sshed to two remote servers, the window title would remain the last one set by the last ssh command.

    The pane title actually show in the right of the status bar. Acorrding to the manpage of tmux (section Name and Titles):

    When a pane is first created, its title is the hostname. A pane's title can be set via the OSC title setting sequence, for example:

          $ printf '\033]2;My Title\033\\'

    See also:

    • command line - Pane Title in Tmux - Stack Overflow

    this_ip=$(ip addr | grep -e '10\.' | awk '{print $2}' | sed -e 's^/2[0-9]$^^' | head -1)

win_title() {
  # set xterm title (and tmux pane title)
  #printf '\033]0;%s\007' $1

win_title "`whoami`@$this_ip"

COLOR_NONE="\[\033[00m\]"
